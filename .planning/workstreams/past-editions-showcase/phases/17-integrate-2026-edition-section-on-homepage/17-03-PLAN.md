```ts
/**
 * SC3 (Phase 17, D-11).
 *
 * Assumes `pnpm build` has run. Asserts homepage heading hierarchy:
 *   - Exactly one <h1> per locale
 *   - No h3 appears before the first h2 (no skipped levels)
 *   - No h4 appears before the first h3
 */
import { describe, it, expect, beforeAll } from "vitest";
import { readFileSync, existsSync } from "node:fs";
import { resolve } from "node:path";

const FILES = [
  ["fr", resolve(import.meta.dirname, "../../dist/index.html")],
  ["en", resolve(import.meta.dirname, "../../dist/en/index.html")],
] as const;

function extractHeadingLevels(html: string): number[] {
  const re = /<h([1-6])\b/gi;
  const out: number[] = [];
  let m: RegExpExecArray | null;
  while ((m = re.exec(html)) !== null) {
    out.push(parseInt(m[1]!, 10));
  }
  return out;
}

describe.each(FILES)("SC3 heading hierarchy — %s", (_locale, file) => {
  let html = "";
  beforeAll(() => {
    if (!existsSync(file)) throw new Error(`Missing ${file}. Run 'pnpm build' first.`);
    html = readFileSync(file, "utf-8");
  });

  it("has exactly one <h1>", () => {
    const levels = extractHeadingLevels(html);
    const h1Count = levels.filter((l) => l === 1).length;
    expect(h1Count, `Expected exactly 1 <h1>, got ${h1Count}`).toBe(1);
  });

  it("does not skip heading levels (no h3 before any h2, no h4 before any h3)", () => {
    const levels = extractHeadingLevels(html);
    let seen = new Set<number>();
    for (const l of levels) {
      if (l === 3) {
        expect(seen.has(2), `<h3> appeared before any <h2>`).toBe(true);
      }
      if (l === 4) {
        expect(seen.has(3), `<h4> appeared before any <h3>`).toBe(true);
      }
      seen.add(l);
    }
  });

  it("includes an <h2> for the 2026 section", () => {
    const levels = extractHeadingLevels(html);
    expect(levels).toContain(2);
  });
});
```
