import { getLangFromUrl, useTranslations, getLocalePath } from "@/i18n/utils";
import LanguageToggle from "@/components/LanguageToggle.astro";
import logoFull from "@/assets/logos/dark/logo.svg";
import logoCompact from "@/assets/logos/dark/logo-notext.svg";

const lang = getLangFromUrl(Astro.url);
const t = useTranslations(lang);
const currentPath = Astro.url.pathname;

const navItems = [
  { key: "nav.home"     as const, path: "/" },
  { key: "nav.speakers" as const, path: "/speakers" },
  { key: "nav.schedule" as const, path: "/" },  // dead link — Phase 7 pending (D-05)
  { key: "nav.sponsors" as const, path: "/" },  // dead link — Phase 5 pending (D-05)
  { key: "nav.venue"    as const, path: "/" },  // dead link — Phase 6 pending (D-05)
  { key: "nav.team"     as const, path: "/" },  // dead link — Phase 5 pending (D-05)
];

const links = navItems.map(({ key, path }) => ({
  key,
  label: t(key),
  href: getLocalePath(lang, path),
}));

function isActive(currentPath: string, href: string): boolean {
  if (href === "/" || href === "/en" || href === "/en/") {
    return currentPath === href || currentPath === href + "/";
  }
  return currentPath.startsWith(href);
}

const homeHref = getLocalePath(lang, "/");
const ctaLabel = t("hero.cta.register");

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL → rendered page | User-controlled URL path drives `getLangFromUrl` and active-link detection. All link hrefs are static-derived — no user input flows into href attributes. |
| i18n key → rendered text | Translation values from `src/i18n/ui.ts` (static build-time data) are interpolated into DOM via Astro's auto-escaping. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-01 | Tampering (XSS) | Navigation.astro — i18n label interpolation | mitigate | Astro auto-escapes `{t(key)}` expressions by default. All nav labels come from static `src/i18n/ui.ts` (build-time constant), not user input. No `set:html` directive used anywhere in Navigation.astro. |
| T-10-02 | Information Disclosure (open redirect) | Navigation.astro — link hrefs | mitigate | All hrefs are either (a) `getLocalePath(lang, staticPath)` where staticPath is a compile-time string literal, or (b) the hardcoded `"#"` placeholder for the CTA. No dynamic href construction from user input. No external URLs in nav. |
| T-10-03 | Tampering (XSS) | Navigation.astro — inline `<script>` | accept | The script uses only `classList.toggle`, `setAttribute("aria-expanded", "true"|"false")`, and `window.scrollY` reads. No innerHTML, no eval, no user-controlled strings. No mitigation beyond existing safe APIs needed. |
| T-10-04 | Tampering (XSS) | TranslationNotice.astro — href via `getLocalePath(originalLocale, currentPath)` | mitigate | `currentPath` comes from `Astro.url.pathname` (server-rendered, not user-injectable at runtime in SSG). `getLocalePath` strips locale segments but does not sanitize — acceptable because output is only used in an `href` attribute that Astro auto-escapes. Non-http(s) URI schemes (`javascript:`) cannot occur because `currentPath` always starts with `/`. |
| T-10-05 | Denial of Service | Navigation.astro — scroll listener | accept | Listener uses `{ passive: true }` and performs only two `classList.toggle` calls. Trivial cost; no throttling needed. |
| T-10-06 | Repudiation / Audit | n/a | accept | Navigation is purely static — no server-side actions, no logging requirements. |
