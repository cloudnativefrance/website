  ```astro
  ---
  import { Font } from "astro:assets";
  import "../styles/global.css";
  import Navigation from "@/components/Navigation.astro";
  import TranslationNotice from "@/components/TranslationNotice.astro";
  import { defaultLang } from "@/i18n/ui";
  import { getLangFromUrl, useTranslations, getLocalePath } from "@/i18n/utils";

  interface Props {
    title?: string;
    description?: string;
    ogImage?: string;
    ogType?: "website" | "event";
    lang?: string;
    canonicalPath?: string;
  }

  const {
    title = "Cloud Native Days France 2027",
    description,
    ogImage = "/og-default.png",
    ogType = "website",
    lang,
    canonicalPath,
  } = Astro.props;

  const resolvedLang = (lang ?? getLangFromUrl(Astro.url)) as "fr" | "en";
  const t = useTranslations(resolvedLang);
  const showTranslationNotice = resolvedLang !== defaultLang;

  // Same-origin OG image allowlist (threat T-09-01): accept only leading-slash paths
  // or absolute URLs starting with Astro.site. Anything else falls back to default.
  const siteOrigin = Astro.site ? Astro.site.origin : "";
  const safeOgImage = (() => {
    if (ogImage.startsWith("/")) return ogImage;
    if (siteOrigin && ogImage.startsWith(siteOrigin)) return ogImage;
    return "/og-default.png";
  })();

  // Absolute URLs
  const currentPath = canonicalPath ?? Astro.url.pathname;
  const canonicalUrl = new URL(currentPath, Astro.site).toString();
  const ogImageUrl = new URL(safeOgImage, Astro.site).toString();

  // hreflang alternate paths — normalize trailing slash, strip locale, re-prefix.
  // Normalizing first ensures /en/privacy/ → /privacy and /en/privacy both emit identical hreflang URLs (plan-checker C1).
  const normalizedPath = currentPath.replace(/\/+$/, "") || "/";
  const pathNoLocale = normalizedPath.replace(/^\/en(\/|$)/, "/") || "/";
  const frPath = getLocalePath("fr", pathNoLocale);
  const enPath = getLocalePath("en", pathNoLocale);
  const frUrl = new URL(frPath, Astro.site).toString();
  const enUrl = new URL(enPath, Astro.site).toString();

  const metaDescription = description ?? t("seo.default_description");
  const ogLocale = resolvedLang === "fr" ? "fr_FR" : "en_US";
  const siteName = t("seo.og_site_name");
  const ogImageAlt = t("seo.og_image_alt");
  ---
  <html lang={resolvedLang} class="dark">
    <head>
      <meta charset="utf-8" />
      <meta name="viewport" content="width=device-width, initial-scale=1" />
      <title>{title}</title>
      <meta name="description" content={metaDescription} />

      <!-- Canonical -->
      <link rel="canonical" href={canonicalUrl} />

      <!-- hreflang alternates -->
      <link rel="alternate" hreflang="fr" href={frUrl} />
      <link rel="alternate" hreflang="en" href={enUrl} />
      <link rel="alternate" hreflang="x-default" href={frUrl} />

      <!-- Open Graph -->
      <meta property="og:type" content={ogType} />
      <meta property="og:title" content={title} />
      <meta property="og:description" content={metaDescription} />
      <meta property="og:image" content={ogImageUrl} />
      <meta property="og:image:alt" content={ogImageAlt} />
      <meta property="og:url" content={canonicalUrl} />
      <meta property="og:locale" content={ogLocale} />
      <meta property="og:site_name" content={siteName} />

      <!-- Twitter card -->
      <meta name="twitter:card" content="summary_large_image" />
      <meta name="twitter:title" content={title} />
      <meta name="twitter:description" content={metaDescription} />
      <meta name="twitter:image" content={ogImageUrl} />

      <!-- Favicon (existing assets) -->
      <link rel="icon" type="image/svg+xml" href="/favicon.svg" />
      <link rel="icon" type="image/x-icon" href="/favicon.ico" />

      <Font cssVariable="--font-dm-sans" preload />
    </head>
    <body class="bg-background text-foreground font-sans antialiased min-h-screen">
      <Navigation />
      {showTranslationNotice && <TranslationNotice />}
      <slot />
    </body>
  </html>
  ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `Astro.props.ogImage` → `<meta og:image>` | Per-page prop controls an absolute URL emitted in HTML; if attacker-controlled (future user-generated content), hostile URLs could mislead social-preview crawlers or point at attacker origins |
| `canonicalPath` prop → `<link rel="canonical">` | Override prop could mis-set canonical if ever fed untrusted input |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-01 | Tampering | Layout.astro `ogImage` prop | mitigate | Same-origin allowlist: accept leading-slash paths OR URLs whose origin matches `Astro.site.origin`. Anything else falls back to `/og-default.png`. Implemented as `safeOgImage` IIFE in Layout frontmatter. |
| T-09-02 | Information Disclosure | Layout.astro `canonicalPath` | accept | All current callers pass either nothing (defaults to `Astro.url.pathname`) or in-repo static strings. No untrusted input reaches this prop in v1.0. If future phases pass dynamic data, this must be revisited. |
| T-09-03 | Spoofing | `hreflang` URL emission | accept | URLs derived from `Astro.url` + `getLocalePath()` — both are framework-controlled; `getLocalePath` strips locale prefixes using the closed `ui` key set. No user input reaches these URLs. |
| T-09-04 | Information Disclosure | `Astro.site` production domain leak in dev | accept | `Astro.site` is public config; leaking it in dev builds is by design. Sitemap + canonical explicitly need absolute URLs. |
</threat_model>
